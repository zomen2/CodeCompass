include_directories(
  include
  ${PROJECT_SOURCE_DIR}/util/include
  ${PROJECT_SOURCE_DIR}/model/include)

  ${ODB_INCLUDE_DIRS})

  src/pluginhandler.cpp

  src/parser.cpp
  src/parsercontext.cpp)

  ${Boost_LINK_LIBRARIES}
  ${ODB_LIBRARIES}
